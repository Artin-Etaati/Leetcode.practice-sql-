create schema if not exists lc608;

drop table if exists lc608.tree;

create table lc608.tree (
    id   int primary key,
    p_id int
);

insert into lc608.tree values
(1, null),
(2, 1),
(3, 1),
(4, 2),
(5, 2);

select * from lc608.tree;

select id,
    case 
        when p_id is null then 'root'
        when id not in (
            select p_id 
            from lc608.tree 
            where p_id is not null
        ) then 'leaf'
        else 'inner'
    end as type
from lc608.tree;
