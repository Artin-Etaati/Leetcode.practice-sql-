CREATE SCHEMA IF NOT EXISTS lc196;

DROP TABLE IF EXISTS lc196.Person;

CREATE TABLE lc196.Person (
    Id    INT,
    Email VARCHAR(50)
);

DELETE FROM lc196.Person
WHERE Id IN (
    SELECT p1.Id
    FROM lc196.Person p1
    JOIN lc196.Person p2
    ON p1.Email = p2.Email
    AND p1.Id > p2.Id
);
