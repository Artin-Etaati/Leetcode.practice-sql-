create schema if not exists lc178;

drop table if exists lc178.scores;

create table lc178.scores (
    id    int primary key,
    score numeric(3,2)
);

insert into lc178.scores values
(1, 3.50),
(2, 3.65),
(3, 4.00),
(4, 3.85),
(5, 4.00),
(6, 3.65);

select * from lc178.scores;

select score, 
dense_rank() over (order by score desc) as rank
from lc178.scores
order by score desc;
