create schema if not exists lc180;

drop table if exists lc180.logs;

create table lc180.logs (
    id  int primary key,
    num int
);

insert into lc180.logs values
(1, 1),
(2, 1),
(3, 1),
(4, 2),
(5, 1),
(6, 2),
(7, 2);

select * from lc180.logs;

select distinct num as consecutivenums
from (
    select num,
        lag(num) over (order by id) as prev_num,
        lead(num) over (order by id) as next_num
    from lc180.logs
) as t
where num = prev_num and num = next_num;
